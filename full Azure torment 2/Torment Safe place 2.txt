repeat task.wait() until game:IsLoaded()

if game.PlaceId == 112802237735950 then
    task.wait(5)

    local player = game.Players.LocalPlayer
    local character = player.Character or player.CharacterAdded:Wait()

    --// Equip Weapons Safely
    local backpack = player:WaitForChild("Backpack")
    local weapon = backpack:WaitForChild("Equinox Slicer")
    if weapon then
        weapon.Parent = character
    end

    --// Started Torment mode
    task.wait(1)
    local skipArgs = { "skipall" }
    game:GetService("ReplicatedStorage"):WaitForChild("Remotes"):WaitForChild("Commands"):FireServer(unpack(skipArgs))

    --// Teleport & Create Safe Platform
    repeat
        task.wait(1)
        if player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
            player.Character:MoveTo(Vector3.new(-950000.23584, 950000.05774, -1651.85693))
            wait()
            player.Character:FindFirstChild("HumanoidRootPart").Anchored = true
        end
    until game:GetService("Workspace"):FindFirstChild("NagaTorment")

    task.wait()
    player.Character:FindFirstChild("HumanoidRootPart").Anchored = false
    --//Just a place
    local rightLeg = character:WaitForChild("Right Leg")
    if rightLeg then
        local NewPart = Instance.new("Part")
        NewPart.Parent = workspace
        NewPart.Size = Vector3.new(1000, 1000.1000, 1000)
        NewPart.BrickColor = BrickColor.new("Blue")
        NewPart.Anchored = true
        NewPart.CFrame = rightLeg.CFrame
    end

    --// Boss Fight
    repeat task.wait() until game:GetService("Workspace"):FindFirstChild("Okeania, the Ocean's Authority")
    task.wait(1)
    
    repeat
        task.wait(0.1)
        if player.Character then
            player.Character:MoveTo(Vector3.new(-950000.23584, 950000.05774, -1651.85693))
        end
    until game:GetService("Workspace"):FindFirstChild("Okeania, the Ocean's Authority") == nil

    --//Tp Stuff
    repeat
        task.wait(0.5)
        local item = backpack:FindFirstChild("Azure Orb")
        if item then
            item.Parent = character
        end
    until character:FindFirstChild("Azure Orb")

    --// Remote for teleporting out
    task.wait()
    local useArgs = { "use", {} }
    character:WaitForChild("Azure Orb"):WaitForChild("WeaponRemote"):FireServer(unpack(useArgs))
    
    task.wait()
    local partyArgs = {
        "createParty",
        {
            settings = {
                Visual = true,
                FriendsOnly = true,
                TormentMode = true
            },
            subplace = "AzureRest"
        }
    }
    game:GetService("ReplicatedStorage"):WaitForChild("Remotes"):WaitForChild("PartySystem"):WaitForChild("PartyFunction"):InvokeServer(unpack(partyArgs))
    
    task.wait()
    local joinArgs = { "joinSubplace", {} }
    game:GetService("ReplicatedStorage"):WaitForChild("Remotes"):WaitForChild("PartySystem"):WaitForChild("PartyFunction"):InvokeServer(unpack(joinArgs))
end
